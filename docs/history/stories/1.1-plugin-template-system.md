---

## Technical Notes

```javascript
// Template structure
module.exports = class PluginTemplate {
  constructor() {
    this.initialized = false;
  }

  start() {
    try {
      this.initialize();
      this.initialized = true;
    } catch (error) {
      console.error(`[Plugin] Error:`, error);
    }
  }

  stop() {
    this.cleanup();
    this.initialized = false;
  }

  getSettingsPanel() {
    return this.buildSettingsUI();
  }
};
```
